#include <iostream>
#include <vector>

using namespace std;

int main(){
    int nums[] = {-1,0,3,5,9,12};
    const int size = 6;
    int target = 12;
    int a = 0, b = size - 1;
        while(a <= b){
            int mid = (a + b) / 2;
            if(nums[mid] == target){
                cout << mid << endl; 
                return 0;
            }
            if(nums[mid] < target){
                a = ++mid;
            }
            else{
                b = --mid;
            }
        }
        cout << -1 << endl;
    return 0;
}
